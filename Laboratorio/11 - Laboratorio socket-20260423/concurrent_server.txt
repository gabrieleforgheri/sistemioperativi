#include <errno.h>
#include <signal.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <sys/un.h>
#include <unistd.h>

#define SOCKET_PATH "/tmp/concurrent.sock"

typedef struct {
    pid_t pid;
    int wait_seconds;
} RequestMsg;

static ssize_t read_full(int fd, void *buf, size_t count) {
    size_t done = 0;
    char *p = buf;

    while (done < count) {
        ssize_t n = read(fd, p + done, count - done);
        if (n == 0) {
            return 0;
        }
        if (n < 0) {
            if (errno == EINTR) {
                continue;
            }
            return -1;
        }
        done += (size_t)n;
    }
    return (ssize_t)done;
}

static void handle_client(int conn_fd) {
    RequestMsg req;
    ssize_t n;

    n = read_full(conn_fd, &req, sizeof(req));
    if (n != (ssize_t)sizeof(req)) {
        if (n < 0) {
            perror("read request");
        } else {
            fprintf(stderr, "[Helper %d] Incomplete request received\n", (int)getpid());
        }
        close(conn_fd);
        _exit(1);
    }

    printf("[Helper %d] Handling request for client pid %d\n", (int)getpid(), (int)req.pid);

    sleep((unsigned int)req.wait_seconds);

    printf("[Helper %d] Handling request for pid %d ended\n", (int)getpid(), (int)req.pid);

    close(conn_fd);
    _exit(0);
}

int main(void) {
    int srv_fd, addr_size;
    struct sockaddr_un srv_addr, cli_addr;
    signal(SIGCHLD, SIG_IGN);

    srv_fd = socket(AF_UNIX, SOCK_STREAM, 0);
    if (srv_fd == -1) {
        perror("socket");
        return 1;
    }

    unlink(SOCKET_PATH);
    memset(&srv_addr, 0, sizeof(srv_addr));
    srv_addr.sun_family = AF_UNIX;
    strncpy(srv_addr.sun_path, SOCKET_PATH, sizeof(srv_addr.sun_path) - 1);

    if (bind(srv_fd, (struct sockaddr *)&srv_addr, sizeof(srv_addr)) == -1) {
        perror("bind");
        close(srv_fd);
        return 1;
    }

    if (listen(srv_fd, 10) == -1) {
        perror("listen");
        close(srv_fd);
        unlink(SOCKET_PATH);
        return 1;
    }

    printf("[Server] Listening on %s\n", SOCKET_PATH);
    fflush(stdout);

    while (1) {
        int conn_fd;
        pid_t child_pid;
	addr_size = sizeof(cli_addr);
        conn_fd = accept(srv_fd, (struct sockaddr *)&cli_addr, &addr_size);
        if (conn_fd == -1) {
            if (errno == EINTR) {
                continue;
            }
            perror("accept");
            break;
        }

        child_pid = fork();
        if (child_pid == -1) {
            perror("fork");
            close(conn_fd);
            continue;
        }

        if (child_pid == 0) {
	    // I am the child that was born to handle this connection
            close(srv_fd);
            handle_client(conn_fd);
        }

	// I am the parent 
        printf("[Server] Accepted connection, helper pid %d created\n", (int)child_pid);
        close(conn_fd);
    }

    close(srv_fd);
    unlink(SOCKET_PATH);
    return 0;
}
