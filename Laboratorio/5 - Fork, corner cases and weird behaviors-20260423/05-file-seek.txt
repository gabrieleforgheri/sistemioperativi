#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <fcntl.h>
#include <sys/wait.h>

int main() {
    int fd = open("data.txt", O_CREAT | O_TRUNC | O_RDWR, 0644);
    if (fd < 0) {
        perror("open");
        exit(1);
    }

    write(fd, "ABCDEFGH", 8);
    lseek(fd, 0, SEEK_SET);

    pid_t pid = fork();
    if (pid < 0) {
        perror("fork");
        exit(1);
    }

    if (pid == 0) {
        lseek(fd, 5, SEEK_SET);   // child moves shared offset
        close(fd);
        exit(0);
    }

    wait(NULL);

    char ch;
    read(fd, &ch, 1);
    printf("Parent read: %c\n", ch);

    close(fd);
    return 0;
}
