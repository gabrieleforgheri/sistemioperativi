#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>

void goodbye(void) {
    printf("Process %d: atexit handler\n", getpid());
}

int main() {
    atexit(goodbye);

    pid_t pid = fork();
    if (pid < 0) {
        perror("fork");
        exit(1);
    }

    if (pid == 0) {
        exit(0);   // replace with _exit(0)
    }

    wait(NULL);
    return 0;
}
